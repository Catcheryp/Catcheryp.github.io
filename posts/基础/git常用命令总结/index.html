//创建分支 
$ git checkout master   将当前分支切换到 master 分支上
$ git checkout -b new-test   将会为你创建一个名为 new-test 的新分支

----
$ git checkout master		//切换到主分支
$ git merge new-test		//合并分支
$ git branch -d new-test    //删除分支
----

$ git branch -l   //查看本地分支
$ git branch -r   //查看远程分支

# 1、当本地和远程的内容不一致，push报错的时候，使用下面的命令可以让，远程的仓库覆盖本地的仓库(如下图)
git fetch --all &&  git reset --hard origin/main && git pull

# 2、更新blog
alias ub="git add . && git commit -m 'update blog' && git push"
# 查看建立的软连接
alias

# 3、git bash中建立永久软连接：新建~/.bashrc文件，并将上面的命令添加进去即可
vim ~/.bashrc
source ~/.bashrc

# 4、git强制提交本地分支覆盖远程分支
git push origin master --force
