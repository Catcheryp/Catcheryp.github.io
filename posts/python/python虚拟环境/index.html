pipenv install

#指定虚拟环境中的python版本

#激活虚拟环境
pipenv shell

#不显式激活虚拟环境，即可在当前项目的虚拟环境中执行命令
pipenv run python  hello.py

Pipfile.lock: 记录了固定版本的详细依赖包列表

virtualenv -p python3 Django

#进入虚拟环境
cd Django

#激活虚拟环境
source ./bin/activate

#退出虚拟环境
deactivate

#删除虚拟环境
rm -rf Django

#带上这个参数，就不会复制任何第三方的包过来
--no-site-packages
